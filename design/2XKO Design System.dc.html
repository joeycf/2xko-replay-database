Tailwind & Token Export

Drop-in for the Nuxt 3 build. Full files also written to /handoff.

tokens.css
:root {
  --bg-base:#0A0B0F;
  --bg-surface:#12141B;
  --bg-elevated:#1A1D26;
  --border:rgba(255,255,255,.11);
  --accent:#FF2E88;
  --accent-hover:#FF529E;
  --accent-2:#38CFFF;
  --text-primary:#F4F5F8;
  --text-muted:#6B7185;
  /* champ slot */
  --champ:#1FE0D4;
  --ease:cubic-bezier(.16,1,.3,1);
}
tailwind.config.js
theme:{extend:{
  colors:{
    base:'#0A0B0F',
    surface:'#12141B',
    accent:{DEFAULT:'#FF2E88',
      hi:'#FF529E'},
  },
  fontFamily:{
    display:['Chakra Petch'],
    sans:['Barlow'],
    mono:['JetBrains Mono'],
  },
  transitionTimingFunction:{
    snap:'cubic-bezier(.16,1,.3,1)'},
}}
